## 问题描述
多个大数相加求和。


## 思路
用数组存储大数，之后模拟运算。

## 代码

```c++
#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
using namespace std;
void revers(char s[])
{
    int c,i,j;
    for(i=0,j=strlen(s)-1;i<j;i++,j--)
    {
        c=s[i];
        s[i]=s[j];
        s[j]=c;
    }
}
int main()
{
    char a[110][110];
    int s[110];
    int line=0,i,j,c,k;
    memset(s,0,sizeof(s));
    while(scanf("%s",a[line]))          //输入大数，并且输入0后退出
    {
        if(strcmp(a[line],"0")==0) 
            break;
        revers(a[line]);
        //printf("%s\n",a[line]);
        line++;
    }
    for(i=0;i<line;i++)
    {
        c=0,k=0;
        for(j=0;j<strlen(a[i]);j++)
        {

            s[k]=s[k]+a[i][j]-'0'+c;
            c=s[k]/10;
            s[k]%=10;
            k++;
        }
        if(c)                   //最高位如果c不为0需进位，因为这个wa
        {
            s[k]+=c;
        }
    }

    for(j=105;j>=0;j--)         //忽略前导0并逆向输出
        if(s[j])
            break;
    for(i=j;i>=0;i--)
        printf("%d",s[i]);
    printf("\n");
    return 0;
}
```
